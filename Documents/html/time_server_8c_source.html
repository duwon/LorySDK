138 void TimerStart( TimerEvent_t *obj )
139 {
140  uint32_t elapsedTime = 0;
141 
142  BACKUP_PRIMASK();
143 
144  DISABLE_IRQ( );
145 
146 
147  if( ( obj == NULL ) || ( TimerExists( obj ) == true ) )
148  {
149  RESTORE_PRIMASK( );
150  return;
151  }
152  obj->Timestamp = obj->ReloadValue;
153  obj->IsRunning = false;
154 
155  if( TimerListHead == NULL )
156  {
157  HW_RTC_SetTimerContext( );
158  TimerInsertNewHeadTimer( obj ); // insert a timeout at now+obj->Timestamp
159  }
160  else
161  {
162  elapsedTime = HW_RTC_GetTimerElapsedTime( );
163  obj->Timestamp += elapsedTime;
164 
165  if( obj->Timestamp < TimerListHead->Timestamp )
166  {
167  TimerInsertNewHeadTimer( obj);
168  }
169  else
170  {
171  TimerInsertTimer( obj);
172  }
173  }
174  RESTORE_PRIMASK( );
175 }

177 static void TimerInsertTimer( TimerEvent_t *obj)
178 {
179  TimerEvent_t* cur = TimerListHead;
180  TimerEvent_t* next = TimerListHead->Next;
181 
182  while (cur->Next != NULL )
183  {
184  if( obj->Timestamp > next->Timestamp )
185  {
186  cur = next;
187  next = next->Next;
188  }
189  else
190  {
191  cur->Next = obj;
192  obj->Next = next;
193  return;
194 
195  }
196  }
197  cur->Next = obj;
198  obj->Next = NULL;
199 }

215 void TimerIrqHandler( void )
216 {
217  TimerEvent_t* cur;
218  TimerEvent_t* next;
219 
220 
221 
222  uint32_t old = HW_RTC_GetTimerContext( );
223  uint32_t now = HW_RTC_SetTimerContext( );
224  uint32_t DeltaContext = now - old; //intentionnal wrap around
225 
226  /* update timeStamp based upon new Time Reference*/
227  /* beacuse delta context should never exceed 2^32*/
228  if ( TimerListHead != NULL )
229  {
230  for (cur=TimerListHead; cur->Next != NULL; cur= cur->Next)
231  {
232  next =cur->Next;
233  if (next->Timestamp > DeltaContext)
234  {
235  next->Timestamp -= DeltaContext;
236  }
237  else
238  {
239  next->Timestamp = 0 ;
240  }
241  }
242  }
243 
244  /* execute imediately the alarm callback */
245  if ( TimerListHead != NULL )
246  {
247  cur = TimerListHead;
248  TimerListHead = TimerListHead->Next;
249  exec_cb( cur->Callback );
250  }
251 
252 
253  // remove all the expired object from the list
254  while( ( TimerListHead != NULL ) && ( TimerListHead->Timestamp < HW_RTC_GetTimerElapsedTime( ) ))
255  {
256  cur = TimerListHead;
257  TimerListHead = TimerListHead->Next;
258  exec_cb( cur->Callback );
259  }
260 
261  /* start the next TimerListHead if it exists AND NOT running */
262  if(( TimerListHead != NULL ) && (TimerListHead->IsRunning == false))
263  {
264  TimerSetTimeout( TimerListHead );
265  }
266 }

268 void TimerStop( TimerEvent_t *obj )
269 {
270  BACKUP_PRIMASK();
271 
272  DISABLE_IRQ( );
273 
274  TimerEvent_t* prev = TimerListHead;
275  TimerEvent_t* cur = TimerListHead;
276 
277  // List is empty or the Obj to stop does not exist
278  if( ( TimerListHead == NULL ) || ( obj == NULL ) )
279  {
280  RESTORE_PRIMASK( );
281  return;
282  }
283 
284  if( TimerListHead == obj ) // Stop the Head
285  {
286  if( TimerListHead->IsRunning == true ) // The head is already running
287  {
288 
289  TimerListHead->IsRunning = false;
290  if( TimerListHead->Next != NULL )
291  {
292  TimerListHead = TimerListHead->Next;
293  TimerSetTimeout( TimerListHead );
294  }
295  else
296  {
297  HW_RTC_StopAlarm( );
298  TimerListHead = NULL;
299  }
300  }
301  else // Stop the head before it is started
302  {
303  if( TimerListHead->Next != NULL )
304  {
305  TimerListHead = TimerListHead->Next;
306  }
307  else
308  {
309  TimerListHead = NULL;
310  }
311  }
312  }
313  else // Stop an object within the list
314  {
315  while( cur != NULL )
316  {
317  if( cur == obj )
318  {
319  if( cur->Next != NULL )
320  {
321  cur = cur->Next;
322  prev->Next = cur;
323  }
324  else
325  {
326  cur = NULL;
327  prev->Next = cur;
328  }
329  break;
330  }
331  else
332  {
333  prev = cur;
334  cur = cur->Next;
335  }
336  }
337  }
338 
339  RESTORE_PRIMASK( );
340 }

363 void TimerSetValue( TimerEvent_t *obj, uint32_t value )
364 {
365  uint32_t minValue = 0;
366  uint32_t ticks = HW_RTC_ms2Tick( value );
367 
368  TimerStop( obj );
369 
370  minValue = HW_RTC_GetMinimumTimeout( );
371 
372  if( ticks < minValue )
373  {
374  ticks = minValue;
375  }
376 
377  obj->Timestamp = ticks;
378  obj->ReloadValue = ticks;
379 }

388 {

390  uint32_t pastInTicks = HW_RTC_ms2Tick( past );

395 static void TimerSetTimeout( TimerEvent_t *obj )
396 {
397  int32_t minTicks= HW_RTC_GetMinimumTimeout( );
398  obj->IsRunning = true;
399 
400  //in case deadline too soon
401  if(obj->Timestamp < (HW_RTC_GetTimerElapsedTime( ) + minTicks) )
402  {
403  obj->Timestamp = HW_RTC_GetTimerElapsedTime( ) + minTicks;
404  }
405  HW_RTC_SetAlarm( obj->Timestamp );
406 }
