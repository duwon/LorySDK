104 static uint8_t CountNbOfEnabledChannels( uint8_t datarate, uint16_t* channelsMask, ChannelParams_t* channels, Band_t* bands, uint8_t* enabledChannels, uint8_t* delayTx )
105 {
106  uint8_t nbEnabledChannels = 0;
107  uint8_t delayTransmission = 0;
108 
109  for( uint8_t i = 0, k = 0; i < CN470_MAX_NB_CHANNELS; i += 16, k++ )
110  {
111  for( uint8_t j = 0; j < 16; j++ )
112  {
113  if( ( channelsMask[k] & ( 1 << j ) ) != 0 )
114  {
115  if( channels[i + j].Frequency == 0 )
116  { // Check if the channel is enabled
117  continue;
118  }
119  if( RegionCommonValueInRange( datarate, channels[i + j].DrRange.Fields.Min,
120  channels[i + j].DrRange.Fields.Max ) == false )
121  { // Check if the current channel selection supports the given datarate
122  continue;
123  }
124  if( bands[channels[i + j].Band].TimeOff > 0 )
125  { // Check if the band is available for transmission
126  delayTransmission++;
127  continue;
128  }
129  enabledChannels[nbEnabledChannels++] = i + j;
130  }
131  }
132  }
133 
134  *delayTx = delayTransmission;
135  return nbEnabledChannels;
136 }

526 uint8_t RegionCN470LinkAdrReq( LinkAdrReqParams_t* linkAdrReq, int8_t* drOut, int8_t* txPowOut, uint8_t* nbRepOut, uint8_t* nbBytesParsed )
527 {
528  uint8_t status = 0x07;
529  RegionCommonLinkAdrParams_t linkAdrParams;
530  uint8_t nextIndex = 0;
531  uint8_t bytesProcessed = 0;
532  uint16_t channelsMask[6] = { 0, 0, 0, 0, 0, 0 };
533  GetPhyParams_t getPhy;
534  PhyParam_t phyParam;
535  RegionCommonLinkAdrReqVerifyParams_t linkAdrVerifyParams;
536 
537  // Initialize local copy of channels mask
538  RegionCommonChanMaskCopy( channelsMask, ChannelsMask, 6 );
539 
540  while( bytesProcessed < linkAdrReq->PayloadSize )
541  {
542  // Get ADR request parameters
543  nextIndex = RegionCommonParseLinkAdrReq( &( linkAdrReq->Payload[bytesProcessed] ), &linkAdrParams );
544 
545  if( nextIndex == 0 )
546  break; // break loop, since no more request has been found
547 
548  // Update bytes processed
549  bytesProcessed += nextIndex;
550 
551  // Revert status, as we only check the last ADR request for the channel mask KO
552  status = 0x07;
553 
554  if( linkAdrParams.ChMaskCtrl == 6 )
555  {
556  // Enable all 125 kHz channels
557  channelsMask[0] = 0xFFFF;
558  channelsMask[1] = 0xFFFF;
559  channelsMask[2] = 0xFFFF;
560  channelsMask[3] = 0xFFFF;
561  channelsMask[4] = 0xFFFF;
562  channelsMask[5] = 0xFFFF;
563  }
564  else if( linkAdrParams.ChMaskCtrl == 7 )
565  {
566  status &= 0xFE; // Channel mask KO
567  }
568  else
569  {
570  for( uint8_t i = 0; i < 16; i++ )
571  {
572  if( ( ( linkAdrParams.ChMask & ( 1 << i ) ) != 0 ) &&
573  ( Channels[linkAdrParams.ChMaskCtrl * 16 + i].Frequency == 0 ) )
574  {// Trying to enable an undefined channel
575  status &= 0xFE; // Channel mask KO
576  }
577  }
578  channelsMask[linkAdrParams.ChMaskCtrl] = linkAdrParams.ChMask;
579  }
580  }
581 
582  // Get the minimum possible datarate
583  getPhy.Attribute = PHY_MIN_TX_DR;
584  getPhy.UplinkDwellTime = linkAdrReq->UplinkDwellTime;
585  phyParam = RegionCN470GetPhyParam( &getPhy );
586 
587  linkAdrVerifyParams.Status = status;
588  linkAdrVerifyParams.AdrEnabled = linkAdrReq->AdrEnabled;
589  linkAdrVerifyParams.Datarate = linkAdrParams.Datarate;
590  linkAdrVerifyParams.TxPower = linkAdrParams.TxPower;
591  linkAdrVerifyParams.NbRep = linkAdrParams.NbRep;
592  linkAdrVerifyParams.CurrentDatarate = linkAdrReq->CurrentDatarate;
593  linkAdrVerifyParams.CurrentTxPower = linkAdrReq->CurrentTxPower;
594  linkAdrVerifyParams.CurrentNbRep = linkAdrReq->CurrentNbRep;
595  linkAdrVerifyParams.NbChannels = CN470_MAX_NB_CHANNELS;
596  linkAdrVerifyParams.ChannelsMask = channelsMask;
597  linkAdrVerifyParams.MinDatarate = ( int8_t )phyParam.Value;
598  linkAdrVerifyParams.MaxDatarate = CN470_TX_MAX_DATARATE;
599  linkAdrVerifyParams.Channels = Channels;
600  linkAdrVerifyParams.MinTxPower = CN470_MIN_TX_POWER;
601  linkAdrVerifyParams.MaxTxPower = CN470_MAX_TX_POWER;
602 
603  // Verify the parameters and update, if necessary
604  status = RegionCommonLinkAdrReqVerifyParams( &linkAdrVerifyParams, &linkAdrParams.Datarate, &linkAdrParams.TxPower, &linkAdrParams.NbRep );
605 
606  // Update channelsMask if everything is correct
607  if( status == 0x07 )
608  {
609  // Copy Mask
610  RegionCommonChanMaskCopy( ChannelsMask, channelsMask, 6 );
611  }
612 
613  // Update status variables
614  *drOut = linkAdrParams.Datarate;
615  *txPowOut = linkAdrParams.TxPower;
616  *nbRepOut = linkAdrParams.NbRep;
617  *nbBytesParsed = bytesProcessed;
618 
619  return status;
620 }
