822  LoRaMacNetID |= ( ( uint32_t )LoRaMacRxPayload[6] << 16 );
823

825  LoRaMacDevAddr |= ( ( uint32_t )LoRaMacRxPayload[8] << 8 );
826  LoRaMacDevAddr |= ( ( uint32_t )LoRaMacRxPayload[9] << 16 );
827  LoRaMacDevAddr |= ( ( uint32_t )LoRaMacRxPayload[10] << 24 );
828

830  LoRaMacParams.Rx1DrOffset = ( LoRaMacRxPayload[11] >> 4 ) & 0x07;
831  LoRaMacParams.Rx2Channel.Datarate = LoRaMacRxPayload[11] & 0x0F;
832 
833  // RxDelay
834  LoRaMacParams.ReceiveDelay1 = ( LoRaMacRxPayload[12] & 0x0F );
835  if( LoRaMacParams.ReceiveDelay1 == 0 )
836  {
837  LoRaMacParams.ReceiveDelay1 = 1;
838  }
839  LoRaMacParams.ReceiveDelay1 *= 1000;
840  LoRaMacParams.ReceiveDelay2 = LoRaMacParams.ReceiveDelay1 + 1000;

845  applyCFList.Size = size - 17;

848 
849  MlmeConfirm.Status = LORAMAC_EVENT_INFO_STATUS_OK;
850  IsLoRaMacNetworkJoined = true;
851  }
852  else
853  {
854  MlmeConfirm.Status = LORAMAC_EVENT_INFO_STATUS_JOIN_FAIL;
855  }
856  break;
857  case FRAME_TYPE_DATA_CONFIRMED_DOWN:
858  case FRAME_TYPE_DATA_UNCONFIRMED_DOWN:
859  {
860  // Check if the received payload size is valid
861  getPhy.UplinkDwellTime = LoRaMacParams.DownlinkDwellTime;
862  getPhy.Datarate = McpsIndication.RxDatarate;
863  getPhy.Attribute = PHY_MAX_PAYLOAD;

869  }
870  phyParam = RegionGetPhyParam( LoRaMacRegion, &getPhy );
871  if( MAX( 0, ( int16_t )( ( int16_t )size - ( int16_t )LORA_MAC_FRMPAYLOAD_OVERHEAD ) ) > phyParam.Value )
872  {
873  McpsIndication.Status = LORAMAC_EVENT_INFO_STATUS_ERROR;
874  PrepareRxDoneAbort( );
875  return;
876  }

878  address = payload[pktHeaderLen++];
879  address |= ( (uint32_t)payload[pktHeaderLen++] << 8 );
880  address |= ( (uint32_t)payload[pktHeaderLen++] << 16 );
881  address |= ( (uint32_t)payload[pktHeaderLen++] << 24 );
882 
883  if( address != LoRaMacDevAddr )
884  {
885  curMulticastParams = MulticastChannels;
886  while( curMulticastParams != NULL )
887  {
888  if( address == curMulticastParams->Address )
889  {
890  multicast = 1;

893  downLinkCounter = curMulticastParams->DownLinkCounter;

899  {

901  McpsIndication.Status = LORAMAC_EVENT_INFO_STATUS_ADDRESS_FAIL;
902  PrepareRxDoneAbort( );
903  return;
904  }

909  nwkSKey = LoRaMacNwkSKey;
910  appSKey = LoRaMacAppSKey;
911  downLinkCounter = DownLinkCounter;

915

923  micRx |= ( ( uint32_t )payload[size - LORAMAC_MFR_LEN + 2] << 16 );
924  micRx |= ( ( uint32_t )payload[size - LORAMAC_MFR_LEN + 3] << 24 );

927  sequenceCounterDiff = ( sequenceCounter - sequenceCounterPrev );
928

930  {

933  if( micRx == mic )
934  {

936  }

1046  {

1623 static LoRaMacStatus_t AddMacCommand( uint8_t cmd, uint8_t p1, uint8_t p2 )
1624 {
1625  LoRaMacStatus_t status = LORAMAC_STATUS_BUSY;
1626  // The maximum buffer length must take MAC commands to re-send into account.
1627  uint8_t bufLen = LORA_MAC_COMMAND_MAX_LENGTH - MacCommandsBufferToRepeatIndex;
1628 
1629  switch( cmd )
1630  {
1631  case MOTE_MAC_LINK_CHECK_REQ:
1632  if( MacCommandsBufferIndex < bufLen )
1633  {
1634  MacCommandsBuffer[MacCommandsBufferIndex++] = cmd;
1635  // No payload for this command
1636  status = LORAMAC_STATUS_OK;
1637  }
1638  break;
1639  case MOTE_MAC_LINK_ADR_ANS:
1640  if( MacCommandsBufferIndex < ( bufLen - 1 ) )
1641  {
1642  MacCommandsBuffer[MacCommandsBufferIndex++] = cmd;
1643  // Margin
1644  MacCommandsBuffer[MacCommandsBufferIndex++] = p1;
1645  status = LORAMAC_STATUS_OK;
1646  }
1647  break;
1648  case MOTE_MAC_DUTY_CYCLE_ANS:
1649  if( MacCommandsBufferIndex < bufLen )
1650  {
1651  MacCommandsBuffer[MacCommandsBufferIndex++] = cmd;
1652  // No payload for this answer
1653  status = LORAMAC_STATUS_OK;
1654  }
1655  break;
1656  case MOTE_MAC_RX_PARAM_SETUP_ANS:
1657  if( MacCommandsBufferIndex < ( bufLen - 1 ) )
1658  {
1659  MacCommandsBuffer[MacCommandsBufferIndex++] = cmd;
1660  // Status: Datarate ACK, Channel ACK
1661  MacCommandsBuffer[MacCommandsBufferIndex++] = p1;
1662  // This is a sticky MAC command answer. Setup indication
1663  SetMlmeScheduleUplinkIndication( );
1664  status = LORAMAC_STATUS_OK;
1665  }
1666  break;
1667  case MOTE_MAC_DEV_STATUS_ANS:
1668  if( MacCommandsBufferIndex < ( bufLen - 2 ) )
1669  {
1670  MacCommandsBuffer[MacCommandsBufferIndex++] = cmd;
1671  // 1st byte Battery
1672  // 2nd byte Margin
1673  MacCommandsBuffer[MacCommandsBufferIndex++] = p1;
1674  MacCommandsBuffer[MacCommandsBufferIndex++] = p2;
1675  status = LORAMAC_STATUS_OK;
1676  }
1677  break;
1678  case MOTE_MAC_NEW_CHANNEL_ANS:
1679  if( MacCommandsBufferIndex < ( bufLen - 1 ) )
1680  {
1681  MacCommandsBuffer[MacCommandsBufferIndex++] = cmd;
1682  // Status: Datarate range OK, Channel frequency OK
1683  MacCommandsBuffer[MacCommandsBufferIndex++] = p1;
1684  status = LORAMAC_STATUS_OK;
1685  }
1686  break;
1687  case MOTE_MAC_RX_TIMING_SETUP_ANS:
1688  if( MacCommandsBufferIndex < bufLen )
1689  {
1690  MacCommandsBuffer[MacCommandsBufferIndex++] = cmd;
1691  // No payload for this answer
1692  // This is a sticky MAC command answer. Setup indication
1693  SetMlmeScheduleUplinkIndication( );
1694  status = LORAMAC_STATUS_OK;
1695  }
1696  break;
1697  case MOTE_MAC_TX_PARAM_SETUP_ANS:
1698  if( MacCommandsBufferIndex < bufLen )
1699  {
1700  MacCommandsBuffer[MacCommandsBufferIndex++] = cmd;
1701  // No payload for this answer
1702  status = LORAMAC_STATUS_OK;
1703  }
1704  break;
1705  case MOTE_MAC_DL_CHANNEL_ANS:
1706  if( MacCommandsBufferIndex < bufLen )
1707  {
1708  MacCommandsBuffer[MacCommandsBufferIndex++] = cmd;
1709  // Status: Uplink frequency exists, Channel frequency OK
1710  MacCommandsBuffer[MacCommandsBufferIndex++] = p1;
1711 
1712  // This is a sticky MAC command answer. Setup indication
1713  SetMlmeScheduleUplinkIndication( );
1714 
1715  status = LORAMAC_STATUS_OK;
1716  }
1717  break;
1718  default:
1719  return LORAMAC_STATUS_SERVICE_UNKNOWN;
1720  }
1721  if( status == LORAMAC_STATUS_OK )

1732

1734  {
1735  return 0;

1771  break;
1772  }

1775  }
1776  }

1781 static void ProcessMacCommands( uint8_t *payload, uint8_t macIndex, uint8_t commandsSize, uint8_t snr )
1782 {
1783  uint8_t status = 0;
1784 
1785  while( macIndex < commandsSize )
1786  {
1787  // Decode Frame MAC commands
1788  switch( payload[macIndex++] )
1789  {
1790  case SRV_MAC_LINK_CHECK_ANS:
1791  MlmeConfirm.Status = LORAMAC_EVENT_INFO_STATUS_OK;
1792  MlmeConfirm.DemodMargin = payload[macIndex++];
1793  MlmeConfirm.NbGateways = payload[macIndex++];
1794  break;
1795  case SRV_MAC_LINK_ADR_REQ:
1796  {
1797  LinkAdrReqParams_t linkAdrReq;
1798  int8_t linkAdrDatarate = DR_0;
1799  int8_t linkAdrTxPower = TX_POWER_0;
1800  uint8_t linkAdrNbRep = 0;
1801  uint8_t linkAdrNbBytesParsed = 0;
1802 
1803  // Fill parameter structure
1804  linkAdrReq.Payload = &payload[macIndex - 1];
1805  linkAdrReq.PayloadSize = commandsSize - ( macIndex - 1 );
1806  linkAdrReq.AdrEnabled = AdrCtrlOn;
1807  linkAdrReq.UplinkDwellTime = LoRaMacParams.UplinkDwellTime;
1808  linkAdrReq.CurrentDatarate = LoRaMacParams.ChannelsDatarate;
1809  linkAdrReq.CurrentTxPower = LoRaMacParams.ChannelsTxPower;
1810  linkAdrReq.CurrentNbRep = LoRaMacParams.ChannelsNbRep;
1811 
1812  // Process the ADR requests
1813  status = RegionLinkAdrReq( LoRaMacRegion, &linkAdrReq, &linkAdrDatarate,
1814  &linkAdrTxPower, &linkAdrNbRep, &linkAdrNbBytesParsed );
1815 
1816  if( ( status & 0x07 ) == 0x07 )
1817  {
1818  LoRaMacParams.ChannelsDatarate = linkAdrDatarate;
1819  LoRaMacParams.ChannelsTxPower = linkAdrTxPower;
1820  LoRaMacParams.ChannelsNbRep = linkAdrNbRep;
1821  }
1822 
1823  // Add the answers to the buffer
1824  for( uint8_t i = 0; i < ( linkAdrNbBytesParsed / 5 ); i++ )
1825  {
1826  AddMacCommand( MOTE_MAC_LINK_ADR_ANS, status, 0 );
1827  }
1828  // Update MAC index
1829  macIndex += linkAdrNbBytesParsed - 1;
1830  }
1831  break;
1832  case SRV_MAC_DUTY_CYCLE_REQ:
1833  MaxDCycle = payload[macIndex++];
1834  AggregatedDCycle = 1 << MaxDCycle;
1835  AddMacCommand( MOTE_MAC_DUTY_CYCLE_ANS, 0, 0 );
1836  break;
1837  case SRV_MAC_RX_PARAM_SETUP_REQ:
1838  {
1839  RxParamSetupReqParams_t rxParamSetupReq;
1840  status = 0x07;
1841 
1842  rxParamSetupReq.DrOffset = ( payload[macIndex] >> 4 ) & 0x07;
1843  rxParamSetupReq.Datarate = payload[macIndex] & 0x0F;
1844  macIndex++;
1845 
1846  rxParamSetupReq.Frequency = ( uint32_t )payload[macIndex++];
1847  rxParamSetupReq.Frequency |= ( uint32_t )payload[macIndex++] << 8;
1848  rxParamSetupReq.Frequency |= ( uint32_t )payload[macIndex++] << 16;
1849  rxParamSetupReq.Frequency *= 100;
1850 
1851  // Perform request on region
1852  status = RegionRxParamSetupReq( LoRaMacRegion, &rxParamSetupReq );
1853 
1854  if( ( status & 0x07 ) == 0x07 )
1855  {
1856  LoRaMacParams.Rx2Channel.Datarate = rxParamSetupReq.Datarate;
1857  LoRaMacParams.Rx2Channel.Frequency = rxParamSetupReq.Frequency;
1858  LoRaMacParams.Rx1DrOffset = rxParamSetupReq.DrOffset;
1859  }
1860  AddMacCommand( MOTE_MAC_RX_PARAM_SETUP_ANS, status, 0 );
1861  }
1862  break;
1863  case SRV_MAC_DEV_STATUS_REQ:
1864  {
1865  uint8_t batteryLevel = BAT_LEVEL_NO_MEASURE;
1866  if( ( LoRaMacCallbacks != NULL ) && ( LoRaMacCallbacks->GetBatteryLevel != NULL ) )
1867  {
1868  batteryLevel = LoRaMacCallbacks->GetBatteryLevel( );
1869  }
1870  AddMacCommand( MOTE_MAC_DEV_STATUS_ANS, batteryLevel, snr & 0x3F);
1871  break;
1872  }
1873  case SRV_MAC_NEW_CHANNEL_REQ:
1874  {
1875  NewChannelReqParams_t newChannelReq;
1876  ChannelParams_t chParam;
1877  status = 0x03;
1878 
1879  newChannelReq.ChannelId = payload[macIndex++];
1880  newChannelReq.NewChannel = &chParam;
1881 
1882  chParam.Frequency = ( uint32_t )payload[macIndex++];
1883  chParam.Frequency |= ( uint32_t )payload[macIndex++] << 8;
1884  chParam.Frequency |= ( uint32_t )payload[macIndex++] << 16;
1885  chParam.Frequency *= 100;
1886  chParam.Rx1Frequency = 0;
1887  chParam.DrRange.Value = payload[macIndex++];
1888 
1889  status = RegionNewChannelReq( LoRaMacRegion, &newChannelReq );
1890 
1891  AddMacCommand( MOTE_MAC_NEW_CHANNEL_ANS, status, 0 );
1892  }
1893  break;
1894  case SRV_MAC_RX_TIMING_SETUP_REQ:
1895  {
1896  uint8_t delay = payload[macIndex++] & 0x0F;
1897 
1898  if( delay == 0 )
1899  {
1900  delay++;
1901  }
1902  LoRaMacParams.ReceiveDelay1 = delay * 1000;
1903  LoRaMacParams.ReceiveDelay2 = LoRaMacParams.ReceiveDelay1 + 1000;
1904  AddMacCommand( MOTE_MAC_RX_TIMING_SETUP_ANS, 0, 0 );
1905  }
1906  break;
1907  case SRV_MAC_TX_PARAM_SETUP_REQ:
1908  {
1909  TxParamSetupReqParams_t txParamSetupReq;
1910  uint8_t eirpDwellTime = payload[macIndex++];
1911 
1912  txParamSetupReq.UplinkDwellTime = 0;
1913  txParamSetupReq.DownlinkDwellTime = 0;
1914 
1915  if( ( eirpDwellTime & 0x20 ) == 0x20 )
1916  {
1917  txParamSetupReq.DownlinkDwellTime = 1;
1918  }
1919  if( ( eirpDwellTime & 0x10 ) == 0x10 )
1920  {
1921  txParamSetupReq.UplinkDwellTime = 1;
1922  }
1923  txParamSetupReq.MaxEirp = eirpDwellTime & 0x0F;
1924 
1925  // Check the status for correctness
1926  if( RegionTxParamSetupReq( LoRaMacRegion, &txParamSetupReq ) != -1 )
1927  {
1928  // Accept command
1929  LoRaMacParams.UplinkDwellTime = txParamSetupReq.UplinkDwellTime;
1930  LoRaMacParams.DownlinkDwellTime = txParamSetupReq.DownlinkDwellTime;
1931  LoRaMacParams.MaxEirp = LoRaMacMaxEirpTable[txParamSetupReq.MaxEirp];
1932  // Add command response
1933  AddMacCommand( MOTE_MAC_TX_PARAM_SETUP_ANS, 0, 0 );
1934  }
1935  }
1936  break;
1937  case SRV_MAC_DL_CHANNEL_REQ:
1938  {
1939  DlChannelReqParams_t dlChannelReq;
1940  status = 0x03;
1941 
1942  dlChannelReq.ChannelId = payload[macIndex++];
1943  dlChannelReq.Rx1Frequency = ( uint32_t )payload[macIndex++];
1944  dlChannelReq.Rx1Frequency |= ( uint32_t )payload[macIndex++] << 8;
1945  dlChannelReq.Rx1Frequency |= ( uint32_t )payload[macIndex++] << 16;
1946  dlChannelReq.Rx1Frequency *= 100;
1947 
1948  status = RegionDlChannelReq( LoRaMacRegion, &dlChannelReq );
1949 
1950  AddMacCommand( MOTE_MAC_DL_CHANNEL_ANS, status, 0 );
1951  }
1952  break;
1953  default:
1954  // Unknown command. ABORT MAC commands processing
1955  return;
1956  }
1957  }
