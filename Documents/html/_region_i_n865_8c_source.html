127

464 bool RegionIN865AdrNext( AdrNextParams_t* adrNext, int8_t* drOut, int8_t* txPowOut, uint32_t* adrAckCounter )
465 {
466  bool adrAckReq = false;
467  int8_t datarate = adrNext->Datarate;
468  int8_t txPower = adrNext->TxPower;
469  GetPhyParams_t getPhy;
470  PhyParam_t phyParam;
471 
472  // Report back the adr ack counter
473  *adrAckCounter = adrNext->AdrAckCounter;
474 
475  if( adrNext->AdrEnabled == true )
476  {
477  if( datarate == IN865_TX_MIN_DATARATE )
478  {
479  *adrAckCounter = 0;
480  adrAckReq = false;
481  }
482  else
483  {
484  if( adrNext->AdrAckCounter >= IN865_ADR_ACK_LIMIT )
485  {
486  adrAckReq = true;
487  txPower = IN865_MAX_TX_POWER;
488  }
489  else
490  {
491  adrAckReq = false;
492  }
493  if( adrNext->AdrAckCounter >= ( IN865_ADR_ACK_LIMIT + IN865_ADR_ACK_DELAY ) )
494  {
495  if( ( adrNext->AdrAckCounter % IN865_ADR_ACK_DELAY ) == 1 )
496  {
497  // Decrease the datarate
498  getPhy.Attribute = PHY_NEXT_LOWER_TX_DR;
499  getPhy.Datarate = datarate;
500  getPhy.UplinkDwellTime = adrNext->UplinkDwellTime;
501  phyParam = RegionIN865GetPhyParam( &getPhy );
502  datarate = phyParam.Value;
503 
504  if( datarate == IN865_TX_MIN_DATARATE )
505  {
506  // We must set adrAckReq to false as soon as we reach the lowest datarate
507  adrAckReq = false;
508  if( adrNext->UpdateChanMask == true )
509  {
510  // Re-enable default channels
511  ChannelsMask[0] |= LC( 1 ) + LC( 2 ) + LC( 3 );
512  }
513  }
514  }
515  }
516  }
517  }
518 
519  *drOut = datarate;
520  *txPowOut = txPower;
521  return adrAckReq;

635 uint8_t RegionIN865LinkAdrReq( LinkAdrReqParams_t* linkAdrReq, int8_t* drOut, int8_t* txPowOut, uint8_t* nbRepOut, uint8_t* nbBytesParsed )
636 {
637  uint8_t status = 0x07;
638  RegionCommonLinkAdrParams_t linkAdrParams;
639  uint8_t nextIndex = 0;
640  uint8_t bytesProcessed = 0;
641  uint16_t chMask = 0;
642  GetPhyParams_t getPhy;
643  PhyParam_t phyParam;
644  RegionCommonLinkAdrReqVerifyParams_t linkAdrVerifyParams;
645 
646  while( bytesProcessed < linkAdrReq->PayloadSize )
647  {
648  // Get ADR request parameters
649  nextIndex = RegionCommonParseLinkAdrReq( &( linkAdrReq->Payload[bytesProcessed] ), &linkAdrParams );
650 
651  if( nextIndex == 0 )
652  break; // break loop, since no more request has been found
653 
654  // Update bytes processed
655  bytesProcessed += nextIndex;
656 
657  // Revert status, as we only check the last ADR request for the channel mask KO
658  status = 0x07;
659 
660  // Setup temporary channels mask
661  chMask = linkAdrParams.ChMask;
662 
663  // Verify channels mask
664  if( ( linkAdrParams.ChMaskCtrl == 0 ) && ( chMask == 0 ) )
665  {
666  status &= 0xFE; // Channel mask KO
667  }
668  else if( ( ( linkAdrParams.ChMaskCtrl >= 1 ) && ( linkAdrParams.ChMaskCtrl <= 5 )) ||
669  ( linkAdrParams.ChMaskCtrl >= 7 ) )
670  {
671  // RFU
672  status &= 0xFE; // Channel mask KO
673  }
674  else
675  {
676  for( uint8_t i = 0; i < IN865_MAX_NB_CHANNELS; i++ )
677  {
678  if( linkAdrParams.ChMaskCtrl == 6 )
679  {
680  if( Channels[i].Frequency != 0 )
681  {
682  chMask |= 1 << i;
683  }
684  }
685  else
686  {
687  if( ( ( chMask & ( 1 << i ) ) != 0 ) &&
688  ( Channels[i].Frequency == 0 ) )
689  {// Trying to enable an undefined channel
690  status &= 0xFE; // Channel mask KO
691  }
692  }
693  }
694  }
695  }
696 
697  if( linkAdrParams.Datarate != DR_6 )
698  {
699  // Get the minimum possible datarate
700  getPhy.Attribute = PHY_MIN_TX_DR;
701  getPhy.UplinkDwellTime = linkAdrReq->UplinkDwellTime;
702  phyParam = RegionIN865GetPhyParam( &getPhy );
703 
704  linkAdrVerifyParams.Status = status;
705  linkAdrVerifyParams.AdrEnabled = linkAdrReq->AdrEnabled;
706  linkAdrVerifyParams.Datarate = linkAdrParams.Datarate;
707  linkAdrVerifyParams.TxPower = linkAdrParams.TxPower;
708  linkAdrVerifyParams.NbRep = linkAdrParams.NbRep;
709  linkAdrVerifyParams.CurrentDatarate = linkAdrReq->CurrentDatarate;
710  linkAdrVerifyParams.CurrentTxPower = linkAdrReq->CurrentTxPower;
711  linkAdrVerifyParams.CurrentNbRep = linkAdrReq->CurrentNbRep;
712  linkAdrVerifyParams.NbChannels = IN865_MAX_NB_CHANNELS;
713  linkAdrVerifyParams.ChannelsMask = &chMask;
714  linkAdrVerifyParams.MinDatarate = ( int8_t )phyParam.Value;
715  linkAdrVerifyParams.MaxDatarate = IN865_TX_MAX_DATARATE;
716  linkAdrVerifyParams.Channels = Channels;
717  linkAdrVerifyParams.MinTxPower = IN865_MIN_TX_POWER;
718  linkAdrVerifyParams.MaxTxPower = IN865_MAX_TX_POWER;
719 
720  // Verify the parameters and update, if necessary
721  status = RegionCommonLinkAdrReqVerifyParams( &linkAdrVerifyParams, &linkAdrParams.Datarate, &linkAdrParams.TxPower, &linkAdrParams.NbRep );
722  }
723  else
724  {// DR_6 is not supported by this region
725  status &= 0xFD; // Datarate KO
726  }
727 
728  // Update channelsMask if everything is correct
729  if( status == 0x07 )
730  {
731  // Set the channels mask to a default value
732  memset1( ( uint8_t* )ChannelsMask, 0, sizeof( ChannelsMask ) );
733  // Update the channels mask
