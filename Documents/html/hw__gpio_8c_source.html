187 static uint8_t HW_GPIO_GetBitPos(uint16_t GPIO_Pin)
188 {
189  uint8_t PinPos=0;
190 
191  if ( ( GPIO_Pin & 0xFF00 ) != 0) { PinPos |= 0x8; }
192  if ( ( GPIO_Pin & 0xF0F0 ) != 0) { PinPos |= 0x4; }
193  if ( ( GPIO_Pin & 0xCCCC ) != 0) { PinPos |= 0x2; }
194  if ( ( GPIO_Pin & 0xAAAA ) != 0) { PinPos |= 0x1; }
195 
196  return PinPos;
197 }
