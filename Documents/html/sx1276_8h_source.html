210 /* Freq = channel * FREQ_STEP */
211 #define SX_CHANNEL_TO_FREQ( channel, freq ) \
212  do \
213  { \
214  uint32_t initialChanInt, initialChanFrac; \
215  initialChanInt = channel >> 8; \
216  initialChanFrac = channel - ( initialChanInt << 8 ); \
217  freq = initialChanInt * FREQ_STEP_8 + ( ( initialChanFrac * FREQ_STEP_8 + ( 128 ) ) >> 8 ); \
218  }while( 0 )
219 
220 /* channel = Freq / FREQ_STEP */
221 #define SX_FREQ_TO_CHANNEL( channel, freq ) \
222  do \
223  { \
224  uint32_t initialFreqInt, initialFreqFrac; \
225  initialFreqInt = freq / FREQ_STEP_8; \
226  initialFreqFrac = freq - ( initialFreqInt * FREQ_STEP_8 ); \
227  channel = ( initialFreqInt << 8 ) + ( ( ( initialFreqFrac << 8 ) + ( FREQ_STEP_8 / 2 ) ) / FREQ_STEP_8 ); \
228  }while( 0 )
