176 void SX1276SetRfTxPower( int8_t power )
177 {
178  uint8_t paConfig = 0;
179  uint8_t paDac = 0;
180 
181  paConfig = SX1276Read( REG_PACONFIG );
182  paDac = SX1276Read( REG_PADAC );
183 
184  paConfig = ( paConfig & RF_PACONFIG_PASELECT_MASK ) | SX1276GetPaSelect( SX1276.Settings.Channel );
185  paConfig = ( paConfig & RF_PACONFIG_MAX_POWER_MASK ) | 0x70;
186 
187  if( ( paConfig & RF_PACONFIG_PASELECT_PABOOST ) == RF_PACONFIG_PASELECT_PABOOST )
188  {
189  if( power > 17 )
190  {
191  paDac = ( paDac & RF_PADAC_20DBM_MASK ) | RF_PADAC_20DBM_ON;
192  }
193  else
194  {
195  paDac = ( paDac & RF_PADAC_20DBM_MASK ) | RF_PADAC_20DBM_OFF;
196  }
197  if( ( paDac & RF_PADAC_20DBM_ON ) == RF_PADAC_20DBM_ON )
198  {
199  if( power < 5 )
200  {
201  power = 5;
202  }
203  if( power > 20 )
204  {
205  power = 20;
206  }
207  paConfig = ( paConfig & RF_PACONFIG_OUTPUTPOWER_MASK ) | ( uint8_t )( ( uint16_t )( power - 5 ) & 0x0F );
208  }
209  else
210  {
211  if( power < 2 )
212  {
213  power = 2;
214  }
215  if( power > 17 )
216  {
217  power = 17;
218  }
219  paConfig = ( paConfig & RF_PACONFIG_OUTPUTPOWER_MASK ) | ( uint8_t )( ( uint16_t )( power - 2 ) & 0x0F );
220  }
221  }
222  else
223  {
224  if( power < -1 )
225  {
226  power = -1;
227  }
228  if( power > 14 )
229  {
230  power = 14;
231  }
232  paConfig = ( paConfig & RF_PACONFIG_OUTPUTPOWER_MASK ) | ( uint8_t )( ( uint16_t )( power + 1 ) & 0x0F );
233  }
234  SX1276Write( REG_PACONFIG, 0x7e );
235  SX1276Write( REG_PADAC, paDac );
236 }
