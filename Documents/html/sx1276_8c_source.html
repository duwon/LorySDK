382 void RxChainCalibration( void )
383 {
384  uint8_t regPaConfigInitVal;
385  uint32_t initialFreq;
386  uint32_t channel;
387 
388  // Save context
389  regPaConfigInitVal = SX1276Read( REG_PACONFIG );
390 
391  channel = ( ( ( uint32_t )SX1276Read( REG_FRFMSB ) << 16 ) |
392  ( ( uint32_t )SX1276Read( REG_FRFMID ) << 8 ) |
393  ( ( uint32_t )SX1276Read( REG_FRFLSB ) ) );
394 
395  SX_CHANNEL_TO_FREQ(channel, initialFreq);
396 
397  // Cut the PA just in case, RFO output, power = -1 dBm
398  SX1276Write( REG_PACONFIG, 0x00 );
399 
400  // Launch Rx chain calibration for LF band
401  SX1276Write( REG_IMAGECAL, ( SX1276Read( REG_IMAGECAL ) & RF_IMAGECAL_IMAGECAL_MASK ) | RF_IMAGECAL_IMAGECAL_START );
402  while( ( SX1276Read( REG_IMAGECAL ) & RF_IMAGECAL_IMAGECAL_RUNNING ) == RF_IMAGECAL_IMAGECAL_RUNNING )
403  {
404  }
405 
406  // Sets a Frequency in HF band
407  SX1276SetChannel( 868000000 );
408 
409  // Launch Rx chain calibration for HF band
410  SX1276Write( REG_IMAGECAL, ( SX1276Read( REG_IMAGECAL ) & RF_IMAGECAL_IMAGECAL_MASK ) | RF_IMAGECAL_IMAGECAL_START );
411  while( ( SX1276Read( REG_IMAGECAL ) & RF_IMAGECAL_IMAGECAL_RUNNING ) == RF_IMAGECAL_IMAGECAL_RUNNING )
412  {
413  }
414 
415  // Restore context
416  SX1276Write( REG_PACONFIG, regPaConfigInitVal );
417  SX1276SetChannel( initialFreq );
418 }

441 void SX1276SetRxConfig( RadioModems_t modem, uint32_t bandwidth,
442  uint32_t datarate, uint8_t coderate,
443  uint32_t bandwidthAfc, uint16_t preambleLen,
444  uint16_t symbTimeout, bool fixLen,
445  uint8_t payloadLen,
446  bool crcOn, bool freqHopOn, uint8_t hopPeriod,
447  bool iqInverted, bool rxContinuous )
448 {
449  SX1276SetModem( modem );
450 
451  switch( modem )
452  {
453  case MODEM_FSK:
454  {
455  SX1276.Settings.Fsk.Bandwidth = bandwidth;
456  SX1276.Settings.Fsk.Datarate = datarate;
457  SX1276.Settings.Fsk.BandwidthAfc = bandwidthAfc;
458  SX1276.Settings.Fsk.FixLen = fixLen;
459  SX1276.Settings.Fsk.PayloadLen = payloadLen;
460  SX1276.Settings.Fsk.CrcOn = crcOn;
461  SX1276.Settings.Fsk.IqInverted = iqInverted;
462  SX1276.Settings.Fsk.RxContinuous = rxContinuous;
463  SX1276.Settings.Fsk.PreambleLen = preambleLen;
464  SX1276.Settings.Fsk.RxSingleTimeout = ( uint32_t )( symbTimeout * ( ( 1.0 / ( double )datarate ) * 8.0 ) * 1000 );
465 
466  datarate = ( uint16_t )( ( double )XTAL_FREQ / ( double )datarate );
467  SX1276Write( REG_BITRATEMSB, ( uint8_t )( datarate >> 8 ) );
468  SX1276Write( REG_BITRATELSB, ( uint8_t )( datarate & 0xFF ) );
469 
470  SX1276Write( REG_RXBW, GetFskBandwidthRegValue( bandwidth ) );
471  SX1276Write( REG_AFCBW, GetFskBandwidthRegValue( bandwidthAfc ) );
472 
473  SX1276Write( REG_PREAMBLEMSB, ( uint8_t )( ( preambleLen >> 8 ) & 0xFF ) );
474  SX1276Write( REG_PREAMBLELSB, ( uint8_t )( preambleLen & 0xFF ) );
475 
476  if( fixLen == 1 )
477  {
478  SX1276Write( REG_PAYLOADLENGTH, payloadLen );
479  }
480  else
481  {
482  SX1276Write( REG_PAYLOADLENGTH, 0xFF ); // Set payload length to the maximum
483  }
484 
485  SX1276Write( REG_PACKETCONFIG1,
486  ( SX1276Read( REG_PACKETCONFIG1 ) &
487  RF_PACKETCONFIG1_CRC_MASK &
488  RF_PACKETCONFIG1_PACKETFORMAT_MASK ) |
489  ( ( fixLen == 1 ) ? RF_PACKETCONFIG1_PACKETFORMAT_FIXED : RF_PACKETCONFIG1_PACKETFORMAT_VARIABLE ) |
490  ( crcOn << 4 ) );
491  SX1276Write( REG_PACKETCONFIG2, ( SX1276Read( REG_PACKETCONFIG2 ) | RF_PACKETCONFIG2_DATAMODE_PACKET ) );
492  }
493  break;
494  case MODEM_LORA:
495  {
496  if( bandwidth > 2 )
497  {
498  // Fatal error: When using LoRa modem only bandwidths 125, 250 and 500 kHz are supported
499  while( 1 );
500  }
501  bandwidth += 7;
502  SX1276.Settings.LoRa.Bandwidth = bandwidth;
503  SX1276.Settings.LoRa.Datarate = datarate;
504  SX1276.Settings.LoRa.Coderate = coderate;
505  SX1276.Settings.LoRa.PreambleLen = preambleLen;
506  SX1276.Settings.LoRa.FixLen = fixLen;
507  SX1276.Settings.LoRa.PayloadLen = payloadLen;
508  SX1276.Settings.LoRa.CrcOn = crcOn;
509  SX1276.Settings.LoRa.FreqHopOn = freqHopOn;
510  SX1276.Settings.LoRa.HopPeriod = hopPeriod;
511  SX1276.Settings.LoRa.IqInverted = iqInverted;
512  SX1276.Settings.LoRa.RxContinuous = rxContinuous;
513 
514  if( datarate > 12 )
515  {
516  datarate = 12;
517  }
518  else if( datarate < 6 )
519  {
520  datarate = 6;
521  }
522 
523  if( ( ( bandwidth == 7 ) && ( ( datarate == 11 ) || ( datarate == 12 ) ) ) ||
524  ( ( bandwidth == 8 ) && ( datarate == 12 ) ) )
525  {
526  SX1276.Settings.LoRa.LowDatarateOptimize = 0x01;
527  }
528  else
529  {
530  SX1276.Settings.LoRa.LowDatarateOptimize = 0x00;
531  }
532 
533  SX1276Write( REG_LR_MODEMCONFIG1,
534  ( SX1276Read( REG_LR_MODEMCONFIG1 ) &
535  RFLR_MODEMCONFIG1_BW_MASK &
536  RFLR_MODEMCONFIG1_CODINGRATE_MASK &
537  RFLR_MODEMCONFIG1_IMPLICITHEADER_MASK ) |
538  ( bandwidth << 4 ) | ( coderate << 1 ) |
539  fixLen );
540 
541  SX1276Write( REG_LR_MODEMCONFIG2,
542  ( SX1276Read( REG_LR_MODEMCONFIG2 ) &
543  RFLR_MODEMCONFIG2_SF_MASK &
544  RFLR_MODEMCONFIG2_RXPAYLOADCRC_MASK &
545  RFLR_MODEMCONFIG2_SYMBTIMEOUTMSB_MASK ) |
546  ( datarate << 4 ) | ( crcOn << 2 ) |
547  ( ( symbTimeout >> 8 ) & ~RFLR_MODEMCONFIG2_SYMBTIMEOUTMSB_MASK ) );
548 
549  SX1276Write( REG_LR_MODEMCONFIG3,
550  ( SX1276Read( REG_LR_MODEMCONFIG3 ) &
551  RFLR_MODEMCONFIG3_LOWDATARATEOPTIMIZE_MASK ) |
552  ( SX1276.Settings.LoRa.LowDatarateOptimize << 3 ) );
553 
554  SX1276Write( REG_LR_SYMBTIMEOUTLSB, ( uint8_t )( symbTimeout & 0xFF ) );
555 
556  SX1276Write( REG_LR_PREAMBLEMSB, ( uint8_t )( ( preambleLen >> 8 ) & 0xFF ) );
557  SX1276Write( REG_LR_PREAMBLELSB, ( uint8_t )( preambleLen & 0xFF ) );
558 
559  if( fixLen == 1 )
560  {
561  SX1276Write( REG_LR_PAYLOADLENGTH, payloadLen );
562  }
563 
564  if( SX1276.Settings.LoRa.FreqHopOn == true )
565  {
566  SX1276Write( REG_LR_PLLHOP, ( SX1276Read( REG_LR_PLLHOP ) & RFLR_PLLHOP_FASTHOP_MASK ) | RFLR_PLLHOP_FASTHOP_ON );
567  SX1276Write( REG_LR_HOPPERIOD, SX1276.Settings.LoRa.HopPeriod );
568  }
569 
570  if( ( bandwidth == 9 ) && ( SX1276.Settings.Channel > RF_MID_BAND_THRESH ) )
571  {
572  // ERRATA 2.1 - Sensitivity Optimization with a 500 kHz Bandwidth
573  SX1276Write( REG_LR_TEST36, 0x02 );
574  SX1276Write( REG_LR_TEST3A, 0x64 );
575  }
576  else if( bandwidth == 9 )
577  {
578  // ERRATA 2.1 - Sensitivity Optimization with a 500 kHz Bandwidth
579  SX1276Write( REG_LR_TEST36, 0x02 );
580  SX1276Write( REG_LR_TEST3A, 0x7F );
581  }
582  else
583  {
584  // ERRATA 2.1 - Sensitivity Optimization with a 500 kHz Bandwidth
585  SX1276Write( REG_LR_TEST36, 0x03 );
586  }
587 
588  if( datarate == 6 )
589  {
590  SX1276Write( REG_LR_DETECTOPTIMIZE,
591  ( SX1276Read( REG_LR_DETECTOPTIMIZE ) &
592  RFLR_DETECTIONOPTIMIZE_MASK ) |
593  RFLR_DETECTIONOPTIMIZE_SF6 );
594  SX1276Write( REG_LR_DETECTIONTHRESHOLD,
595  RFLR_DETECTIONTHRESH_SF6 );
596  }
597  else
598  {
599  SX1276Write( REG_LR_DETECTOPTIMIZE,
600  ( SX1276Read( REG_LR_DETECTOPTIMIZE ) &
601  RFLR_DETECTIONOPTIMIZE_MASK ) |
602  RFLR_DETECTIONOPTIMIZE_SF7_TO_SF12 );
603  SX1276Write( REG_LR_DETECTIONTHRESHOLD,
604  RFLR_DETECTIONTHRESH_SF7_TO_SF12 );
605  }
606  }
607  break;
608  }
609 }

744 uint32_t SX1276GetTimeOnAir( RadioModems_t modem, uint8_t pktLen )
745 {

748  switch( modem )

756  pktLen +
757  ( ( SX1276.Settings.Fsk.CrcOn == 0x01 ) ? 2.0 : 0 ) ) /
758  SX1276.Settings.Fsk.Datarate ) * 1000 );
759  }
760  break;
761  case MODEM_LORA:
762  {
763  double bw = 0.0;
764  // REMARK: When using LoRa modem only bandwidths 125, 250 and 500 kHz are supported
765  switch( SX1276.Settings.LoRa.Bandwidth )
766  {
767  //case 0: // 7.8 kHz
768  // bw = 7800;
769  // break;
770  //case 1: // 10.4 kHz
771  // bw = 10400;
772  // break;
773  //case 2: // 15.6 kHz

775  // break;
776  //case 3: // 20.8 kHz
777  // bw = 20800;
778  // break;
779  //case 4: // 31.2 kHz
780  // bw = 31200;
781  // break;
782  //case 5: // 41.4 kHz
783  // bw = 41400;
784  // break;
785  //case 6: // 62.5 kHz
786  // bw = 62500;
787  // break;
788  case 7: // 125 kHz

793  break;
794  case 9: // 500 kHz
795  bw = 500000;
796  break;
797  }
798 
799  // Symbol rate : time for one symbol (secs)
800  double rs = bw / ( 1 << SX1276.Settings.LoRa.Datarate );
801  double ts = 1 / rs;
802  // time of preamble
803  double tPreamble = ( SX1276.Settings.LoRa.PreambleLen + 4.25 ) * ts;
804  // Symbol length of payload and time

806  28 + 16 * SX1276.Settings.LoRa.CrcOn -
807  ( SX1276.Settings.LoRa.FixLen ? 20 : 0 ) ) /
808  ( double )( 4 * ( SX1276.Settings.LoRa.Datarate -
809  ( ( SX1276.Settings.LoRa.LowDatarateOptimize > 0 ) ? 2 : 0 ) ) ) ) *
810  ( SX1276.Settings.LoRa.Coderate + 4 );
811  double nPayload = 8 + ( ( tmp > 0 ) ? tmp : 0 );
812  double tPayload = nPayload * ts;
813  // Time on air
814  double tOnAir = tPreamble + tPayload;
815  // return ms secs
816  airTime = (uint32_t) floor( tOnAir * 1000 + 0.999 );
817  }
818  break;
819  }
820  return airTime;
