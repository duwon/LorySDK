71 void vcom_Init(void)
72 {
73  /*## Configure the UART peripheral ######################################*/
74  /* Put the USART peripheral in the Asynchronous mode (UART Mode) */
75  /* UART1 configured as follow:
76  - Word Length = 8 Bits
77  - Stop Bit = One Stop bit
78  - Parity = ODD parity
79  - BaudRate = 921600 baud
80  - Hardware flow control disabled (RTS and CTS signals) */
81  UartHandle.Instance = USARTX;
82 
83  UartHandle.Init.BaudRate = 115200;
84  UartHandle.Init.WordLength = UART_WORDLENGTH_8B;
85  UartHandle.Init.StopBits = UART_STOPBITS_1;
86  UartHandle.Init.Parity = UART_PARITY_NONE;
87  UartHandle.Init.HwFlowCtl = UART_HWCONTROL_NONE;
88  UartHandle.Init.Mode = UART_MODE_TX_RX;
89 
90  if(HAL_UART_Init(&UartHandle) != HAL_OK)
91  {
92  /* Initialization Error */
93  Error_Handler();
94  }
95 
96  HAL_NVIC_SetPriority(USARTX_IRQn, 0x1, 0);
97  HAL_NVIC_EnableIRQ(USARTX_IRQn);
98 }

108 void vcom_Send( char *format, ... )
109 {
110  va_list args;
111  va_start(args, format);
112  uint8_t len;
113  uint8_t lenTop;
114  char tempBuff[128];
115 
116  BACKUP_PRIMASK();
117  DISABLE_IRQ();
118 
119  /*convert into string at buff[0] of length iw*/
120  len = vsprintf(&tempBuff[0], format, args);
121 
122  if (iw+len<BUFSIZE)
123  {
124  memcpy( &buff[iw], &tempBuff[0], len);
125  iw+=len;
126  }
127  else
128  {
129  lenTop=BUFSIZE-iw;
130  memcpy( &buff[iw], &tempBuff[0], lenTop);
131  len-=lenTop;
132  memcpy( &buff[0], &tempBuff[lenTop], len);
133  iw = len;
134  }
135  RESTORE_PRIMASK();
136 
137  HAL_NVIC_SetPendingIRQ(USARTX_IRQn);
138 
139  va_end(args);
140 }

143 void vcom_Print( void)
144 {
145  char* CurChar;
146  while( ( (iw+BUFSIZE-ir)%BUFSIZE) >0 )
147  {
148  BACKUP_PRIMASK();
149  DISABLE_IRQ();
150 
151  CurChar = &buff[ir];
152  ir= (ir+1) %BUFSIZE;
153 
154  RESTORE_PRIMASK();
155 
156  HAL_UART_Transmit(&UartHandle,(uint8_t *) CurChar, 1, 300);
157  }
158  HAL_NVIC_ClearPendingIRQ(USARTX_IRQn);
159 }
