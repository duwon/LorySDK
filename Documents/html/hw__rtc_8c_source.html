252 void HW_RTC_setMcuWakeUpTime( void )
253 {
254  RTC_TimeTypeDef RTC_TimeStruct;
255  RTC_DateTypeDef RTC_DateStruct;
256 
257  TimerTime_t now, hit;
258  int16_t McuWakeUpTime;
259 
260  if ((McuWakeUpTimeInitialized == false) &&
261  ( HAL_NVIC_GetPendingIRQ( RTC_Alarm_IRQn ) == 1))
262  { /* warning: works ok if now is below 30 days
263  it is ok since it's done once at first alarm wake-up*/
264  McuWakeUpTimeInitialized = true;
265  now = HW_RTC_GetCalendarValue( &RTC_DateStruct, &RTC_TimeStruct );
266 
267  HAL_RTC_GetAlarm(&RtcHandle, &RTC_AlarmStructure, RTC_ALARM_A, RTC_FORMAT_BIN );
268  hit = RTC_AlarmStructure.AlarmTime.Seconds+
269  60*(RTC_AlarmStructure.AlarmTime.Minutes+
270  60*(RTC_AlarmStructure.AlarmTime.Hours+
271  24*(RTC_AlarmStructure.AlarmDateWeekDay)));
272  hit = ( hit << N_PREDIV_S ) + (PREDIV_S - RTC_AlarmStructure.AlarmTime.SubSeconds);
273 
274  McuWakeUpTime = (int16_t) ((now-hit));
275  McuWakeUpTimeCal += McuWakeUpTime;
276  DBG_PRINTF("Cal=%d, %d\n\r",McuWakeUpTimeCal, McuWakeUpTime);
277  }
278 }

322 void HW_RTC_SetAlarm( uint32_t timeout )
323 {
324  /* we don't go in Low Power mode for timeout below MIN_ALARM_DELAY */
325  if ( (MIN_ALARM_DELAY + McuWakeUpTimeCal ) < ((timeout - HW_RTC_GetTimerElapsedTime( ) )) )
326  {
327  LPM_SetStopMode(LPM_RTC_Id , LPM_Enable );
328  }
329  else
330  {
331  LPM_SetStopMode(LPM_RTC_Id , LPM_Disable );
332  }
333 
334  /*In case stop mode is required */
335  if( LPM_GetMode() == LPM_StopMode )
336  {
337  timeout = timeout - McuWakeUpTimeCal;
338  }
339 
340  HW_RTC_StartWakeUpAlarm( timeout );
341 }

394 {
395  RTC_HandleTypeDef* hrtc=&RtcHandle;
396  /* enable low power at irq*/
397  LPM_SetStopMode(LPM_RTC_Id , LPM_Enable );
398 
399  /* Clear the EXTI's line Flag for RTC Alarm */
400  __HAL_RTC_ALARM_EXTI_CLEAR_FLAG();
401 
402  /* Get the AlarmA interrupt source enable status */
403  if(__HAL_RTC_ALARM_GET_IT_SOURCE(hrtc, RTC_IT_ALRA) != RESET)
404  {
405  /* Get the pending status of the AlarmA Interrupt */
406  if(__HAL_RTC_ALARM_GET_FLAG(hrtc, RTC_FLAG_ALRAF) != RESET)
407  {
408  /* Clear the AlarmA interrupt pending bit */
409  __HAL_RTC_ALARM_CLEAR_FLAG(hrtc, RTC_FLAG_ALRAF);
410  /* AlarmA callback */
411  HAL_RTC_AlarmAEventCallback(hrtc);
412  }

475 static void HW_RTC_StartWakeUpAlarm( uint32_t timeoutValue )
476 {
477  uint16_t rtcAlarmSubSeconds = 0;
478  uint16_t rtcAlarmSeconds = 0;
479  uint16_t rtcAlarmMinutes = 0;
480  uint16_t rtcAlarmHours = 0;
481  uint16_t rtcAlarmDays = 0;
482  RTC_TimeTypeDef RTC_TimeStruct = RtcTimerContext.RTC_Calndr_Time;
483  RTC_DateTypeDef RTC_DateStruct = RtcTimerContext.RTC_Calndr_Date;
484 
485  HW_RTC_StopAlarm( );
486 
487 
488  /*reverse counter */
489  rtcAlarmSubSeconds = PREDIV_S - RTC_TimeStruct.SubSeconds;
490  rtcAlarmSubSeconds += ( timeoutValue & PREDIV_S);
491  /* convert timeout to seconds */
492  timeoutValue >>= N_PREDIV_S; /* convert timeout in seconds */
493 
494  /*convert microsecs to RTC format and add to 'Now' */
495  rtcAlarmDays = RTC_DateStruct.Date;
496  while (timeoutValue >= SECONDS_IN_1DAY)
497  {
498  timeoutValue -= SECONDS_IN_1DAY;
499  rtcAlarmDays++;
500  }
501 
502  /* calc hours */
503  rtcAlarmHours = RTC_TimeStruct.Hours;
504  while (timeoutValue >= SECONDS_IN_1HOUR)
505  {
506  timeoutValue -= SECONDS_IN_1HOUR;
507  rtcAlarmHours++;
508  }
509 
510  /* calc minutes */
511  rtcAlarmMinutes = RTC_TimeStruct.Minutes;
512  while (timeoutValue >= SECONDS_IN_1MINUTE)
513  {
514  timeoutValue -= SECONDS_IN_1MINUTE;
515  rtcAlarmMinutes++;
516  }
517 
518  /* calc seconds */
519  rtcAlarmSeconds = RTC_TimeStruct.Seconds + timeoutValue;
520 
521  /***** correct for modulo********/
522  while (rtcAlarmSubSeconds >= (PREDIV_S+1))
523  {
524  rtcAlarmSubSeconds -= (PREDIV_S+1);
525  rtcAlarmSeconds++;
526  }
527 
528  while (rtcAlarmSeconds >= SECONDS_IN_1MINUTE)
529  {
530  rtcAlarmSeconds -= SECONDS_IN_1MINUTE;
531  rtcAlarmMinutes++;
532  }
533 
534  while (rtcAlarmMinutes >= MINUTES_IN_1HOUR)
535  {
536  rtcAlarmMinutes -= MINUTES_IN_1HOUR;
537  rtcAlarmHours++;
538  }
539 
540  while (rtcAlarmHours >= HOURS_IN_1DAY)
541  {
542  rtcAlarmHours -= HOURS_IN_1DAY;
543  rtcAlarmDays++;
544  }
545 
546  if( RTC_DateStruct.Year % 4 == 0 )
547  {
548  if( rtcAlarmDays > DaysInMonthLeapYear[ RTC_DateStruct.Month - 1 ] )
549  {
550  rtcAlarmDays = rtcAlarmDays % DaysInMonthLeapYear[ RTC_DateStruct.Month - 1 ];
551  }
552  }
553  else
554  {
555  if( rtcAlarmDays > DaysInMonth[ RTC_DateStruct.Month - 1 ] )
556  {
557  rtcAlarmDays = rtcAlarmDays % DaysInMonth[ RTC_DateStruct.Month - 1 ];
558  }
559  }
560 
561  /* Set RTC_AlarmStructure with calculated values*/
562  RTC_AlarmStructure.AlarmTime.SubSeconds = PREDIV_S-rtcAlarmSubSeconds;
563  RTC_AlarmStructure.AlarmSubSecondMask = HW_RTC_ALARMSUBSECONDMASK;
564  RTC_AlarmStructure.AlarmTime.Seconds = rtcAlarmSeconds;
565  RTC_AlarmStructure.AlarmTime.Minutes = rtcAlarmMinutes;
566  RTC_AlarmStructure.AlarmTime.Hours = rtcAlarmHours;
567  RTC_AlarmStructure.AlarmDateWeekDay = ( uint8_t )rtcAlarmDays;
568  RTC_AlarmStructure.AlarmTime.TimeFormat = RTC_TimeStruct.TimeFormat;
569  RTC_AlarmStructure.AlarmDateWeekDaySel = RTC_ALARMDATEWEEKDAYSEL_DATE;
570  RTC_AlarmStructure.AlarmMask = RTC_ALARMMASK_NONE;
571  RTC_AlarmStructure.Alarm = RTC_ALARM_A;
572  RTC_AlarmStructure.AlarmTime.DayLightSaving = RTC_DAYLIGHTSAVING_NONE;
573  RTC_AlarmStructure.AlarmTime.StoreOperation = RTC_STOREOPERATION_RESET;
574 
575  /* Set RTC_Alarm */
576  HAL_RTC_SetAlarm_IT( &RtcHandle, &RTC_AlarmStructure, RTC_FORMAT_BIN );
577 }

586 static TimerTime_t HW_RTC_GetCalendarValue( RTC_DateTypeDef* RTC_DateStruct, RTC_TimeTypeDef* RTC_TimeStruct )
587 {
588  TimerTime_t calendarValue = 0;
589  uint32_t first_read;
590  uint32_t correction;
591 
592  /* Get Time and Date*/
593  HAL_RTC_GetTime( &RtcHandle, RTC_TimeStruct, RTC_FORMAT_BIN );
594 
595  /* make sure it is correct due to asynchronus nature of RTC*/
596  do {
597  first_read = RTC_TimeStruct->SubSeconds;
598  HAL_RTC_GetDate( &RtcHandle, RTC_DateStruct, RTC_FORMAT_BIN );
599  HAL_RTC_GetTime( &RtcHandle, RTC_TimeStruct, RTC_FORMAT_BIN );
600  } while (first_read != RTC_TimeStruct->SubSeconds);
601 
602  /* calculte amount of elapsed days since 01/01/2000 */
603  calendarValue= DIVC( (DAYS_IN_YEAR*3 + DAYS_IN_LEAP_YEAR)* RTC_DateStruct->Year , 4);
604 
605  correction = ( (RTC_DateStruct->Year % 4) == 0 ) ? DAYS_IN_MONTH_CORRECTION_LEAP : DAYS_IN_MONTH_CORRECTION_NORM ;
606 
607  calendarValue +=( DIVC( (RTC_DateStruct->Month-1)*(30+31) ,2 ) - (((correction>> ((RTC_DateStruct->Month-1)*2) )&0x3)));
608 
609  calendarValue += (RTC_DateStruct->Date -1);
610 
611  /* convert from days to seconds */
612  calendarValue *= SECONDS_IN_1DAY;
613 
614  calendarValue += ( ( uint32_t )RTC_TimeStruct->Seconds +
615  ( ( uint32_t )RTC_TimeStruct->Minutes * SECONDS_IN_1MINUTE ) +
616  ( ( uint32_t )RTC_TimeStruct->Hours * SECONDS_IN_1HOUR ) ) ;
617 
618 
619 
620  calendarValue = (calendarValue<<N_PREDIV_S) + ( PREDIV_S - RTC_TimeStruct->SubSeconds);
621 
622  return( calendarValue );
623 }
