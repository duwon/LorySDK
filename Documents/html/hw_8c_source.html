142 void HW_GpioInit(void)
143 {
144  GPIO_InitTypeDef GPIO_InitStruct={0};
145 
146  /* Configure all GPIO as analog to reduce current consumption on non used IOs */
147  /* Enable GPIOs clock */
148  __HAL_RCC_GPIOA_CLK_ENABLE();
149  __HAL_RCC_GPIOB_CLK_ENABLE();
150  __HAL_RCC_GPIOC_CLK_ENABLE();
151  __HAL_RCC_GPIOD_CLK_ENABLE();
152  __HAL_RCC_GPIOH_CLK_ENABLE();
153 
154  GPIO_InitStruct.Mode = GPIO_MODE_ANALOG;
155  GPIO_InitStruct.Pull = GPIO_NOPULL;
156  /* All GPIOs except debug pins (SWCLK and SWD) */
157  GPIO_InitStruct.Pin = GPIO_PIN_All & (~( GPIO_PIN_13 | GPIO_PIN_14) );
158  HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);
159 
160  /* All GPIOs */
161  GPIO_InitStruct.Pin = GPIO_PIN_All;
162  HAL_GPIO_Init(GPIOB, &GPIO_InitStruct);
163  HAL_GPIO_Init(GPIOC, &GPIO_InitStruct);
164  HAL_GPIO_Init(GPIOD, &GPIO_InitStruct);
165  HAL_GPIO_Init(GPIOH, &GPIO_InitStruct);
166 
167  /* Disable GPIOs clock */
168  __HAL_RCC_GPIOA_CLK_DISABLE();
169  __HAL_RCC_GPIOB_CLK_DISABLE();
170  __HAL_RCC_GPIOC_CLK_DISABLE();
171  __HAL_RCC_GPIOD_CLK_DISABLE();
172  __HAL_RCC_GPIOH_CLK_DISABLE();
173 }

191 void SystemClock_Config( void )
192 {
193  RCC_ClkInitTypeDef RCC_ClkInitStruct = {0};
194  RCC_OscInitTypeDef RCC_OscInitStruct = {0};
195 
196  /* Enable HSE Oscillator and Activate PLL with HSE as source */
197  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSI;
198  RCC_OscInitStruct.HSIState = RCC_HSI_ON;
199  RCC_OscInitStruct.HSICalibrationValue = RCC_HSICALIBRATION_DEFAULT;
200  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
201  RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSI;
202  RCC_OscInitStruct.PLL.PLLMUL = RCC_PLL_MUL6;
203  RCC_OscInitStruct.PLL.PLLDIV = RCC_PLL_DIV3;
204 
205  if (HAL_RCC_OscConfig(&RCC_OscInitStruct) != HAL_OK)
206  {
207  Error_Handler();
208  }
209 
210  /* Set Voltage scale1 as MCU will run at 32MHz */
211  __HAL_RCC_PWR_CLK_ENABLE();
212  __HAL_PWR_VOLTAGESCALING_CONFIG(PWR_REGULATOR_VOLTAGE_SCALE1);
213 
214  /* Poll VOSF bit of in PWR_CSR. Wait until it is reset to 0 */
215  while (__HAL_PWR_GET_FLAG(PWR_FLAG_VOS) != RESET) {};
216 
217  /* Select PLL as system clock source and configure the HCLK, PCLK1 and PCLK2
218  clocks dividers */
219  RCC_ClkInitStruct.ClockType = (RCC_CLOCKTYPE_SYSCLK | RCC_CLOCKTYPE_HCLK | RCC_CLOCKTYPE_PCLK1 | RCC_CLOCKTYPE_PCLK2);
220  RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
221  RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
222  RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV1;
223  RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV1;
224  if (HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_1) != HAL_OK)
225  {
226  Error_Handler();
227  }
228 }

264 {

266

268 
269  HW_IoDeInit( );
270

272  SET_BIT(PWR->CR, PWR_CR_CWUF);
273

285 void LPM_ExitStopMode( void)
286 {
287  /* Disable IRQ while the MCU is not running on HSI */
288 
289  BACKUP_PRIMASK();
290 
291  DISABLE_IRQ( );
292 
293  /* After wake-up from STOP reconfigure the system clock */
294  /* Enable HSI */
295  __HAL_RCC_HSI_ENABLE();
296 
297  /* Wait till HSI is ready */
298  while( __HAL_RCC_GET_FLAG(RCC_FLAG_HSIRDY) == RESET ) {}
299 
300  /* Enable PLL */
301  __HAL_RCC_PLL_ENABLE();
302  /* Wait till PLL is ready */
303  while( __HAL_RCC_GET_FLAG( RCC_FLAG_PLLRDY ) == RESET ) {}
304 
305  /* Select PLL as system clock source */
306  __HAL_RCC_SYSCLK_CONFIG ( RCC_SYSCLKSOURCE_PLLCLK );
307 
308  /* Wait till PLL is used as system clock source */
309  while( __HAL_RCC_GET_SYSCLK_SOURCE( ) != RCC_SYSCLKSOURCE_STATUS_PLLCLK ) {}
310 
311  /*initilizes the peripherals*/
312  HW_IoInit( );
313 
314  RESTORE_PRIMASK( );
315 }
