407 bool LORA_send(lora_AppData_t* AppData, LoraConfirm_t IsTxConfirmed)
408 {
409  McpsReq_t mcpsReq;
410  LoRaMacTxInfo_t txInfo;
411 
412  /*if certification test are on going, application data is not sent*/
413  if (certif_running() == true)
414  {
415  return false;
416  }
417 
418  if( LoRaMacQueryTxPossible( AppData->BuffSize, &txInfo ) != LORAMAC_STATUS_OK )
419  {
420  // Send empty frame in order to flush MAC commands
421  mcpsReq.Type = MCPS_UNCONFIRMED;
422  mcpsReq.Req.Unconfirmed.fBuffer = NULL;
423  mcpsReq.Req.Unconfirmed.fBufferSize = 0;
424  mcpsReq.Req.Unconfirmed.Datarate = LoRaParamInit->TxDatarate;
425  }
426  else
427  {
428  if( IsTxConfirmed == LORAWAN_UNCONFIRMED_MSG )
429  {
430  mcpsReq.Type = MCPS_UNCONFIRMED;
431  mcpsReq.Req.Unconfirmed.fPort = AppData->Port;
432  mcpsReq.Req.Unconfirmed.fBufferSize = AppData->BuffSize;
433  mcpsReq.Req.Unconfirmed.fBuffer = AppData->Buff;
434  mcpsReq.Req.Unconfirmed.Datarate = LoRaParamInit->TxDatarate;
435  }
436  else
437  {
438  mcpsReq.Type = MCPS_CONFIRMED;
439  mcpsReq.Req.Confirmed.fPort = AppData->Port;
440  mcpsReq.Req.Confirmed.fBufferSize = AppData->BuffSize;
441  mcpsReq.Req.Confirmed.fBuffer = AppData->Buff;
442  mcpsReq.Req.Confirmed.NbTrials = 8;
443  mcpsReq.Req.Confirmed.Datarate = LoRaParamInit->TxDatarate;
444  }
445  }
446  if( LoRaMacMcpsRequest( &mcpsReq ) == LORAMAC_STATUS_OK )
447  {
448  return false;
449  }
450  return true;
451 }

453 LoraErrorStatus LORA_RequestClass( DeviceClass_t newClass )
454 {
455  LoraErrorStatus Errorstatus = LORA_SUCCESS;
456  MibRequestConfirm_t mibReq;
457  DeviceClass_t currentClass;
458 
459  mibReq.Type = MIB_DEVICE_CLASS;
460  LoRaMacMibGetRequestConfirm( &mibReq );
461 
462  currentClass = mibReq.Param.Class;
463  /*attempt to swicth only if class update*/
464  if (currentClass != newClass)
465  {
466  switch (newClass)
467  {
468  case CLASS_A:
469  {
470  if (currentClass == CLASS_A)
471  {
472  mibReq.Param.Class = CLASS_A;
473  if( LoRaMacMibSetRequestConfirm( &mibReq ) == LORAMAC_STATUS_OK )
474  {
475  /*switch is instantanuous*/
476  LoRaMainCallbacks->LORA_ConfirmClass(CLASS_A);
477  }
478  else
479  {
480  Errorstatus = LORA_ERROR;
481  }
482  }
483  break;
484  }
485  case CLASS_C:
486  {
487  if (currentClass != CLASS_A)
488  {
489  Errorstatus = LORA_ERROR;
490  }
491  /*switch is instantanuous*/
492  mibReq.Param.Class = CLASS_C;
493  if( LoRaMacMibSetRequestConfirm( &mibReq ) == LORAMAC_STATUS_OK )
494  {
495  LoRaMainCallbacks->LORA_ConfirmClass(CLASS_C);
496  }
497  else
498  {
499  Errorstatus = LORA_ERROR;
500  }
501  break;
502  }
503  default:
504  break;
505  }
506  }
507  return Errorstatus;
508 }
