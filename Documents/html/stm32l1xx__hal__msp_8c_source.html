92 void HAL_MspInit(void)

95 
96  /* Disable the Power Voltage Detector */
97  HAL_PWR_DisablePVD( );
98 
99  /* Set MCU in ULP (Ultra Low Power) */

111  HW_GpioInit( );

128 void HAL_RTC_MspInit(RTC_HandleTypeDef *hrtc)
129 {
130  RCC_OscInitTypeDef RCC_OscInitStruct;
131  RCC_PeriphCLKInitTypeDef PeriphClkInitStruct;
132 
133  /*##-1- Configue the RTC clock soucre ######################################*/
134  /* -a- Enable LSE Oscillator */
135  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_LSE;
136  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_NONE;
137  RCC_OscInitStruct.LSEState = RCC_LSE_ON;
138  if(HAL_RCC_OscConfig(&RCC_OscInitStruct) != HAL_OK)
139  {
140  Error_Handler();
141  }
142 
143  /* -b- Select LSI as RTC clock source */
144  PeriphClkInitStruct.PeriphClockSelection = RCC_PERIPHCLK_RTC;
145  PeriphClkInitStruct.RTCClockSelection = RCC_RTCCLKSOURCE_LSE;
146  if(HAL_RCCEx_PeriphCLKConfig(&PeriphClkInitStruct) != HAL_OK)
147  {
148  Error_Handler();
149  }
150 
151  /*##-2- Enable the RTC peripheral Clock ####################################*/
152  /* Enable RTC Clock */
153  __HAL_RCC_RTC_ENABLE();
154 
155  /*##-3- Configure the NVIC for RTC Alarm ###################################*/
156  HAL_NVIC_SetPriority(RTC_Alarm_IRQn, 0x0, 0);
157  HAL_NVIC_EnableIRQ(RTC_Alarm_IRQn);
158 }
