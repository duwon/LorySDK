200 static uint32_t SpiFrequency( uint32_t hz )
201 {
202  uint32_t divisor = 0;
203  uint32_t SysClkTmp = SystemCoreClock;
204  uint32_t baudRate;
205 
206  while( SysClkTmp > hz)
207  {
208  divisor++;
209  SysClkTmp= ( SysClkTmp >> 1);
210 
211  if (divisor >= 7)
212  break;
213  }
214 
215  baudRate =((( divisor & 0x4 ) == 0 )? 0x0 : SPI_CR1_BR_2 )|
216  ((( divisor & 0x2 ) == 0 )? 0x0 : SPI_CR1_BR_1 )|
217  ((( divisor & 0x1 ) == 0 )? 0x0 : SPI_CR1_BR_0 );
218 
219  return baudRate;
220 }
