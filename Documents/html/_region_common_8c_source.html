162 TimerTime_t RegionCommonUpdateBandTimeOff( bool joined, bool dutyCycle, Band_t* bands, uint8_t nbBands )
163 {
164  TimerTime_t nextTxDelay = ( TimerTime_t )( -1 );
165 
166  // Update bands Time OFF
167  for( uint8_t i = 0; i < nbBands; i++ )
168  {
169  if( joined == false )
170  {
171  uint32_t txDoneTime = MAX( TimerGetElapsedTime( bands[i].LastJoinTxDoneTime ),
172  ( dutyCycle == true ) ? TimerGetElapsedTime( bands[i].LastTxDoneTime ) : 0 );
173 
174  if( bands[i].TimeOff <= txDoneTime )
175  {
176  bands[i].TimeOff = 0;
177  }
178  if( bands[i].TimeOff != 0 )
179  {
180  nextTxDelay = MIN( bands[i].TimeOff - txDoneTime, nextTxDelay );
181  }
182  }
183  else
184  {
185  if( dutyCycle == true )
186  {
187  if( bands[i].TimeOff <= TimerGetElapsedTime( bands[i].LastTxDoneTime ) )
188  {
189  bands[i].TimeOff = 0;
190  }
191  if( bands[i].TimeOff != 0 )
192  {
193  nextTxDelay = MIN( bands[i].TimeOff - TimerGetElapsedTime( bands[i].LastTxDoneTime ),
194  nextTxDelay );
195  }
196  }
197  else
198  {
199  nextTxDelay = 0;
200  bands[i].TimeOff = 0;
201  }
202  }
203  }

207 uint8_t RegionCommonParseLinkAdrReq( uint8_t* payload, RegionCommonLinkAdrParams_t* linkAdrParams )
208 {
209  uint8_t retIndex = 0;
210 
211  if( payload[0] == SRV_MAC_LINK_ADR_REQ )
212  {
213  // Parse datarate and tx power
214  linkAdrParams->Datarate = payload[1];
215  linkAdrParams->TxPower = linkAdrParams->Datarate & 0x0F;
216  linkAdrParams->Datarate = ( linkAdrParams->Datarate >> 4 ) & 0x0F;
217  // Parse ChMask
218  linkAdrParams->ChMask = ( uint16_t )payload[2];
219  linkAdrParams->ChMask |= ( uint16_t )payload[3] << 8;
220  // Parse ChMaskCtrl and nbRep
221  linkAdrParams->NbRep = payload[4];
222  linkAdrParams->ChMaskCtrl = ( linkAdrParams->NbRep >> 4 ) & 0x07;
223  linkAdrParams->NbRep &= 0x0F;
224 
225  // LinkAdrReq has 4 bytes length + 1 byte CMD
226  retIndex = 5;
227  }
228  return retIndex;
229 }

231 uint8_t RegionCommonLinkAdrReqVerifyParams( RegionCommonLinkAdrReqVerifyParams_t* verifyParams, int8_t* dr, int8_t* txPow, uint8_t* nbRep )
232 {
233  uint8_t status = verifyParams->Status;
234  int8_t datarate = verifyParams->Datarate;
235  int8_t txPower = verifyParams->TxPower;
236  int8_t nbRepetitions = verifyParams->NbRep;
237 
238  // Handle the case when ADR is off.
239  if( verifyParams->AdrEnabled == false )
240  {
241  // When ADR is off, we are allowed to change the channels mask and the NbRep,
242  // if the datarate and the TX power of the LinkAdrReq are set to 0x0F.
243  if( ( verifyParams->Datarate != 0x0F ) || ( verifyParams->TxPower != 0x0F ) )
244  {
245  status = 0;
246  nbRepetitions = verifyParams->CurrentNbRep;
247  }
248  // Get the current datarate and tx power
249  datarate = verifyParams->CurrentDatarate;
250  txPower = verifyParams->CurrentTxPower;
251  }
252 
253  if( status != 0 )
254  {
255  // Verify datarate. The variable phyParam. Value contains the minimum allowed datarate.
256  if( RegionCommonChanVerifyDr( verifyParams->NbChannels, verifyParams->ChannelsMask, datarate,
257  verifyParams->MinDatarate, verifyParams->MaxDatarate, verifyParams->Channels ) == false )
258  {
259  status &= 0xFD; // Datarate KO
260  }
261 
262  // Verify tx power
263  if( RegionCommonValueInRange( txPower, verifyParams->MaxTxPower, verifyParams->MinTxPower ) == 0 )
264  {
265  // Verify if the maximum TX power is exceeded
266  if( verifyParams->MaxTxPower > txPower )
267  { // Apply maximum TX power. Accept TX power.
268  txPower = verifyParams->MaxTxPower;
269  }
270  else
271  {
272  status &= 0xFB; // TxPower KO
273  }
274  }
275  }
276 
277  // If the status is ok, verify the NbRep
278  if( status == 0x07 )
279  {
280  if( nbRepetitions == 0 )
281  { // Restore the default value according to the LoRaWAN specification
282  nbRepetitions = 1;
283  }
284  }
285 
286  // Apply changes
287  *dr = datarate;
288  *txPow = txPower;
289  *nbRep = nbRepetitions;
290 
291  return status;
292 }

319 void RegionCommonCalcBackOff( RegionCommonCalcBackOffParams_t* calcBackOffParams )
320 {
321  uint8_t bandIdx = calcBackOffParams->Channels[calcBackOffParams->Channel].Band;
322  uint16_t dutyCycle = calcBackOffParams->Bands[bandIdx].DCycle;
323  uint16_t joinDutyCycle = 0;
324 
325  // Reset time-off to initial value.
326  calcBackOffParams->Bands[bandIdx].TimeOff = 0;
327 
328  if( calcBackOffParams->Joined == false )
329  {
330  // Get the join duty cycle
331  joinDutyCycle = RegionCommonGetJoinDc( calcBackOffParams->ElapsedTime );
332  // Apply the most restricting duty cycle
333  dutyCycle = MAX( dutyCycle, joinDutyCycle );
334  // Reset the timeoff if the last frame was not a join request and when the duty cycle is not enabled
335  if( ( calcBackOffParams->DutyCycleEnabled == false ) && ( calcBackOffParams->LastTxIsJoinRequest == false ) )
336  {
337  // This is the case when the duty cycle is off and the last uplink frame was not a join.
338  // This could happen in case of a rejoin, e.g. in compliance test mode.
339  // In this special case we have to set the time off to 0, since the join duty cycle shall only
340  // be applied after the first join request.
341  calcBackOffParams->Bands[bandIdx].TimeOff = 0;
342  }
343  else
344  {
345  // Apply band time-off.
346  calcBackOffParams->Bands[bandIdx].TimeOff = calcBackOffParams->TxTimeOnAir * dutyCycle - calcBackOffParams->TxTimeOnAir;
347  }
348  }
349  else
350  {
351  if( calcBackOffParams->DutyCycleEnabled == true )
352  {
353  calcBackOffParams->Bands[bandIdx].TimeOff = calcBackOffParams->TxTimeOnAir * dutyCycle - calcBackOffParams->TxTimeOnAir;
354  }
355  else
356  {
357  calcBackOffParams->Bands[bandIdx].TimeOff = 0;
358  }
359  }
360 }
