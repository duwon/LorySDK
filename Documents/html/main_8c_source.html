151 static void Send( void )
152 {
153  uint16_t pressure = 900;
154  int16_t temperature = 30;
155  uint16_t humidity = 80;
156 
157  if ( LORA_JoinStatus () != LORA_SET) /*Not joined, try again later*/
158  {
159  LORA_Join();
160  return;
161  }
162 
163  uint8_t cchannel=0;
164  uint32_t i = 0;
165 
166  AppData.Port = LPP_APP_PORT;
167 
168  AppData.Buff[i++] = cchannel++;
169  AppData.Buff[i++] = LPP_DATATYPE_BAROMETER;
170  AppData.Buff[i++] = ( pressure >> 8 ) & 0xFF;
171  AppData.Buff[i++] = pressure & 0xFF;
172  AppData.Buff[i++] = cchannel++;
173  AppData.Buff[i++] = LPP_DATATYPE_TEMPERATURE;
174  AppData.Buff[i++] = ( temperature >> 8 ) & 0xFF;
175  AppData.Buff[i++] = temperature & 0xFF;
176  AppData.Buff[i++] = cchannel++;
177  AppData.Buff[i++] = LPP_DATATYPE_HUMIDITY;
178  AppData.Buff[i++] = humidity & 0xFF;
179 
180  AppData.BuffSize = i;
181 
182  LORA_send( &AppData, LORAWAN_DEFAULT_CONFIRM_MSG_STATE);
183 
184 }

187 static void LORA_RxData( lora_AppData_t *AppData )
188 {
189 
190  PRINTF("%d PACKET RECEIVED ON PORT %d\n\r", HW_RTC_GetTimerValue(), AppData->Port);
191 
192  switch (AppData->Port)
193  {
194  case 3:
195  /*this port switches the class*/
196  if( AppData->BuffSize == 1 )
197  {
198  switch ( AppData->Buff[0] )
199  {
200  case 0:
201  {
202  LORA_RequestClass(CLASS_A);
203  break;
204  }
205  case 1:
206  {
207  LORA_RequestClass(CLASS_B);

210  case 2:
211  {
212  LORA_RequestClass(CLASS_C);
213  break;
214  }
215  default:
216  break;
217  }
218  }
219  break;
220  case LORAWAN_APP_PORT:
221  if( AppData->BuffSize == 1 )
222  {
223  AppLedStateOn = AppData->Buff[0] & 0x01;
224  if ( AppLedStateOn == RESET )
