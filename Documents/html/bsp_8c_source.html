135  gpioinitstruct.Pull = GPIO_NOPULL;
136  gpioinitstruct.Speed = GPIO_SPEED_FREQ_MEDIUM;
137 
138  if (ButtonMode == BUTTON_MODE_GPIO)
139  {
140  /* Configure Button pin as input */
141  gpioinitstruct.Mode = GPIO_MODE_INPUT;
142 
143  HAL_GPIO_Init(BUTTON_PORT[Button], &gpioinitstruct);
144  }
145 
146  if (ButtonMode == BUTTON_MODE_EXTI)
147  {
148  /* Configure Button pin as input with External interrupt */
149  gpioinitstruct.Mode = GPIO_MODE_IT_FALLING;
150  HAL_GPIO_Init(BUTTON_PORT[Button], &gpioinitstruct);
151 
152  /* Enable and set Button EXTI Interrupt to the lowest priority */
153  HAL_NVIC_SetPriority((IRQn_Type)(BUTTON_IRQn[Button]), 0x0F, 0);
154  HAL_NVIC_EnableIRQ((IRQn_Type)(BUTTON_IRQn[Button]));
155  }
